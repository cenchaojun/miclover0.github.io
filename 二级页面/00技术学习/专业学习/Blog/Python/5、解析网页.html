import requestsfrom bs4 import BeautifulSoup

link = "http://www.santostang.com/"headers = {'User-Agent' : 'Mozilla/5.0 (Windows; U; Windows NT 6.1; en-US; rv:1.9.1.6) Gecko/20091201 Firefox/3.5.6'}
r = requests.get(link, headers= headers)

soup = BeautifulSoup(r.text,"html.parser")
first_title = soup.find("h1", class_="post-title").a.text.strip()#获取单个标题名
print ("第一篇文章的标题是：", first_title)

title_list = soup.find_all("h1", class_="post-title")#获取所有标题名
for i in range(len(title_list)):
title = title_list[i].a.text.strip()
print ('第 %s 篇文章的标题是：%s' %(i+1, title))
