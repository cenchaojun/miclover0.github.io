      f = open("C:/Users/JIA/Desktop/数据结构/file.txt", "r")

      lines = f.readlines() # 读取全部内容

       html = BeautifulSoup(html, 'html.parser')

       attrs = html.a.attrs

       if attrs['data-type'] == "ppt":

       print(attrs['data-title'] + attrs['data-type'], attrs['data-url'])

       else:

       print(attrs['data-title'], attrs['data-url'])
